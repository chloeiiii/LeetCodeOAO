## Course Schedule - Medium
https://leetcode.com/problems/course-schedule/description/

There are a total of numCourses courses you have to take, labeled from 0 to numCourses - 1. You are given an array prerequisites where prerequisites[i] = [ai, bi] indicates that you must take course bi first if you want to take course ai.

For example, the pair [0, 1], indicates that to take course 0 you have to first take course 1.
Return true if you can finish all courses. Otherwise, return false.

  

Example 1:

    Input: numCourses = 2, prerequisites = [[1,0]]
    Output: true
    Explanation: There are a total of 2 courses to take. 
    To take course 1 you should have finished course 0. So it is possible.
Example 2:

    Input: numCourses = 2, prerequisites = [[1,0],[0,1]]
    Output: false
    Explanation: There are a total of 2 courses to take. 
    To take course 1 you should have finished course 0, and to take course 0 you should also have finished course 1. So it is impossible.


Constraints:

    1 <= numCourses <= 2000
    0 <= prerequisites.length <= 5000
    prerequisites[i].length == 2
    0 <= ai, bi < numCourses
    All the pairs prerequisites[i] are unique.

    //DFS cycle detection
    class Solution {
        HashMap<Integer, ArrayList<Integer>> map = new HashMap<Integer, ArrayList<Integer>>();
        boolean[] verified;
        boolean[] visited;
        public boolean canFinish(int numCourses, int[][] prerequisites) {
            // If a cycle exists -> false
            // Build the map
            // course -> prerequisites
            // e.g. 1 -> [0, 2, 3]
            for(int[] pair:prerequisites){
                int key = pair[0];
                int val = pair[1];
                ArrayList<Integer> precourses = map.getOrDefault(key, new ArrayList<Integer>());
                precourses.add(val);
                map.put(key, precourses);
            }
    
    
            verified = new boolean[numCourses];
            visited = new boolean[numCourses];
    
            for(int course=0; course<numCourses; course++){
                if(verified[course]!=true){
                    if(hasCycle(course, visited)){
                        return false;
                    } 
                }
            }
            return true;
        }
        
        public boolean hasCycle(int course, boolean[] visited){
            //this couse has been verified
            if(verified[course]) return false;
    
            //this course does not contain any prerequisites 
            if(!map.containsKey(course)){
                verified[course] = true;
                return false;
            }
    
            //found the course in the current path -> cycle
            if(visited[course] == true) return true;
    
            //add current course to visited array
            visited[course] = true;
    
            //continue add this course's prerequisites courses to the visited array
            ArrayList<Integer> precourses =  map.get(course);
            for(int precourse: precourses){
                if(hasCycle(precourse, visited)){
                    return true;
                }
            }
    
            verified[course] = true;
            //remove current course
            visited[course] = false;
    
            return false;
        }
    }

      //Topological Sort BFS
      class Solution {
          HashMap<Integer, ArrayList<Integer>> map = new HashMap<Integer, ArrayList<Integer>>();
          int[] indegree;
          public boolean canFinish(int numCourses, int[][] prerequisites) {
              //how many prerequisites this course still needs
              indegree = new int[numCourses]; 
              // Build the map
              // prereq-> courses that depend on it
              // e.g. 
              // 0 -> [1, 2]
              // 1 -> [3]
              // 2 -> [3]
              for(int[] pair:prerequisites){
                  int course = pair[0];
                  int precourse = pair[1];
                  ArrayList<Integer> courses = map.getOrDefault(precourse, new ArrayList<Integer>());
                  courses.add(course);
                  map.put(precourse, courses);
                  indegree[course] += 1;
              }
      
              //add all course with 0 prerequisites to the queue
              Queue<Integer> queue = new ArrayDeque<Integer>();
              for(int i=0; i<numCourses; i++){
                  if(indegree[i] == 0){
                      queue.offer(i);
                  }
              }
      
              int completed = 0;
              while(!queue.isEmpty()){
                  int curCourse = queue.poll();
                  completed ++;
      
                  if(map.containsKey(curCourse)){
                      ArrayList<Integer> courses = map.get(curCourse);
                      for(int c : courses){
                          indegree[c] --;
                          if(indegree[c] == 0){
                              queue.offer(c);
                          }
                      }
                  }
              }
              return completed == numCourses;
          }
      }
